Mockapitron

Tool for generating random api response based on

online version http://api-mockapitron.rhcloud.com/

Usage example:

fetch('http://api-mockapitron.rhcloud.com/', {
  method: 'post',
  headers: {
    'Accept': 'application/json',
    'Content-Type': 'application/json'
  },
  mode: "cors", 
  body: JSON.stringify({
    schema: {
      name: 'name',
      adress: 'address'
    },
    size: 3
  })
})
.then(function(res){ return res.json() })
.then(function(data){ console.log(data) })
